Sprite Class

Defined in: src\are\display\Sprite.js:1

精灵，继承自DisplayObject，具体的使用例子如下：

        var sprite = new Sprite({
           x: 200,
           y: 200,
           framerate: 5,
           imgs: [ld.get("hero"), ld.get("pig")],
           frames: [
                   // x, y, width, height, imageIndex, originX, originY               
                   [64, 64, 64, 64],
                   [128, 64, 64, 64],
                   [192, 64, 64, 64],
                   [256, 64, 64, 64],
                   [320, 64, 64, 64],
                   [384, 64, 64, 64],
                   [448, 64, 64, 64],
                       [0, 192, 64, 64],
                       [64, 192, 64, 64],
                   [128, 192, 64, 64],
                   [192, 192, 64, 64],
                   [256, 192, 64, 64],
                   [320, 192, 64, 64],
                   [384, 192, 64, 64],
                   [448, 192, 64, 64],
                       [448, 192, 64, 64]
           ],
           animations: {
               walk: {
                   frames: [0, 1, 2, 3, 4, 5, 6],
                   next: "run",
                   speed: 2,
                   loop: false
               },
               happy: {
                   frames: [11, 12, 13, 14]
               },
               win: {
                   frames: [7, 8, 9, 10]
               }
           },
           currentAnimation: "walk",
           tickAnimationEnd: function () {
               console.log("播完一轮");
           },
           animationEnd: function () {
               alert("播放完成")
           }
       });
       stage.add(sprite);
